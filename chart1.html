#!/usr/bin/env python3 """ chart3_creation.py — Testing Efficiency Table (Marxx brand colors) Reads chart_3.json → generates a single interactive Plotly table. """ import json import plotly.graph_objects as go # ── Load data ────────────────────────────────────────────────────────────────── with open("/Users/nishant/Documents/scripts/report_2026/chart_3.json") as f: data = json.load(f) rows = data["rows"] headers = data["headers"] # ── Marxx AI brand palette ───────────────────────────────────────────────────── TIER_COLORS = { "Micro (<$10K)": "#93C5FD", "Small ($10K-$50K)": "#3B82F6", "Medium ($50K-$200K)": "#6366F1", "Large ($200K-$1M)": "#E879A8", "Enterprise ($1M+)": "#F59E0B", } BG_COLOR = "#111827" # Dark Grey / Slate 900 HEADER_BG = "#1F2937" # Slate 800 ROW_EVEN_BG = "#111827" ROW_ODD_BG = "#1F2937" TEXT_COLOR = "#F9FAFB" # Light Text HEADER_TEXT = "#F9FAFB" LINE_COLOR = "#374151" # ── Prepare Table Data ───────────────────────────────────────────────────────── col_spend_tier = [r["spendTier"] for r in rows] col_volume = [r["averageTestingVolumePerWeek"] for r in rows] col_hit_rate = [f"{r['averageHitRatePercent']}%" for r in rows] # Header labels (replacing \n with
for HTML rendering in Plotly) header_labels = [h["label"].replace("\n", "
") for h in headers] # ── Create Plotly Table ──────────────────────────────────────────────────────── fig = go.Figure(data=[go.Table( header=dict( values=header_labels, fill_color=HEADER_BG, align='left', font=dict(color=HEADER_TEXT, size=14, family="Inter, system-ui, sans-serif"), height=50 ), cells=dict( values=[col_spend_tier, col_volume, col_hit_rate], fill_color=[ # Column 1 colors (Tier colored chips) [BG_COLOR] * len(rows), # Other columns background [ROW_EVEN_BG if i % 2 == 0 else ROW_ODD_BG for i in range(len(rows))], [ROW_EVEN_BG if i % 2 == 0 else ROW_ODD_BG for i in range(len(rows))] ], align='left', font=dict(color=TEXT_COLOR, size=13, family="Inter, system-ui, sans-serif"), height=40, line_color="#E2E8F0" ) )]) # Add tier-colored indicator circles or just style the Spend Tier text # Plotly Table is a bit limited for complex cell-level HTML, # so we'll use a better approach: Colored backgrounds for the first column. fig.data[0].cells.fill.color = [ [TIER_COLORS[t] for t in col_spend_tier], # Col 1 gets tier colors [ROW_EVEN_BG if i % 2 == 0 else ROW_ODD_BG for i in range(len(rows))], # Col 2 [ROW_EVEN_BG if i % 2 == 0 else ROW_ODD_BG for i in range(len(rows))] # Col 3 ] # Adjust font color for Col 1 (light text on dark-ish tiers, dark on light tiers) fig.data[0].cells.font.color = [ ["#FFFFFF" if t != "Micro (<$10K)" else TEXT_COLOR for t in col_spend_tier], TEXT_COLOR, TEXT_COLOR ] fig.update_layout( title=dict( text="Average Testing Efficiency by Spend Tier", font=dict(size=22, color=TEXT_COLOR, family="Inter, system-ui, sans-serif"), x=0.5, ), paper_bgcolor=BG_COLOR, margin=dict(l=20, r=20, t=80, b=20), height=400 ) # ── Output HTML ──────────────────────────────────────────────────────────────── html_content = f""" Testing Efficiency Table — Marxx
{fig.to_html(full_html=False, include_plotlyjs=False)}
""" output_path = "/Users/nishant/Documents/scripts/report_2026/chart3_dashboard.html" with open(output_path, "w") as f: f.write(html_content) print(f"✅ Dashboard saved to: {output_path}")
